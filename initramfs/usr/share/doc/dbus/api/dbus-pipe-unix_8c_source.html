47_dbus_pipe_write (DBusPipe *pipe,

51 DBusError *error)
52{
53 int written;
54
55 written = _dbus_write (pipe->fd, buffer, start, len);
56 if (written < 0)
57 {
58 dbus_set_error (error, DBUS_ERROR_FAILED,
59 "Writing to pipe: %s\n",
60 _dbus_strerror (errno));
61 }
62 return written;
63}
