46struct DBusString
47{
48#if defined(DBUS_WIN) && defined(_DEBUG)
49 const char *dummy1;
50#else
51 const void *dummy1;
52#endif
53 int dummy2;
54 int dummy3;

56 unsigned int dummy_bit2 : 1;
57 unsigned int dummy_bit3 : 1;
58 unsigned int dummy_bits : 3;
59};
